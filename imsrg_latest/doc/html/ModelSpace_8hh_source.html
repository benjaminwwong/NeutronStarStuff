45 struct Orbit
46 {
47  int n;
48  int l;
49  int j2;
50  int tz2;
51  double occ; // particle=0, hole=1
52  double occ_nat; // occupation in the natural orbital basis (if we use natural orbitals) particle=0, hole=1
53  int cvq; // core=0, valence=1, qspace=2
54  int index;
55 
56  Orbit(){};
57  Orbit(int n, int l, int j2, int tz2, double occ, int cvq, int index)
58  : n(n), l(l), j2(j2), tz2(tz2),occ(occ),occ_nat(occ),cvq(cvq),index(index) {};
59 
60  Orbit(const Orbit& orb) : n(orb.n), l(orb.l), j2(orb.j2), tz2(orb.tz2),occ(orb.occ),occ_nat(orb.occ),cvq(orb.cvq),index(orb.index) {}
61  bool operator==( const Orbit& rhs ) const { return ( n==rhs.n and l==rhs.l and j2==rhs.j2 and tz2==rhs.tz2 ); };
62 };

67 struct Ket // | pq >
68 {
69  // Fields
70  Orbit* op;
71  Orbit* oq;
72  size_t p;
73  size_t q;
74  int dpq;
75  int phase_prefactor;
76 
77  // Methods
78  Ket(){};
79  Ket(Orbit& op_in, Orbit& oq_in) : op(&op_in), oq(&oq_in), p(op_in.index), q(oq_in.index), dpq( op_in.index==oq_in.index ? 1 : 0)
80  {
81  phase_prefactor = ((op->j2+oq->j2)/2 + 1) % 2==0 ? 1 : -1;
82  };
83 
84  int Phase(int J) { return phase_prefactor * (J%2==0 ? 1 : -1); };
85  int delta_pq() const {return dpq;};
86 
87 };
